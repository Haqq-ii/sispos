      Import SpeechClient from '@google-cloud/speech'
      const speechClient = new SpeechClient()  // reads GOOGLE_APPLICATION_CREDENTIALS from env automatically
      transcribeAudio(audioBuffer: Buffer): Promise<string>
        Call speechClient.recognize with:
          audio: { content: audioBuffer.toString('base64') }
          config:
            encoding: 'WEBM_OPUS'           // WebM/Opus from MediaRecorder
            sampleRateHertz: 48000           // default MediaRecorder WebM/Opus
            languageCode: 'id-ID'            // CLAUDE.md §STT
            model: 'latest_long'             // best accuracy for conversational Indonesian
            enableAutomaticPunctuation: true
        Join results: response.results?.map(r => r.alternatives?.[0]?.transcript ?? '').join(' ').trim() ?? ''
        Return transcript string (empty string if no speech detected)

      Handle case where GOOGLE_APPLICATION_CREDENTIALS is not set:
        if (!process.env.GOOGLE_APPLICATION_CREDENTIALS) {
          logger.warn('GOOGLE_APPLICATION_CREDENTIALS not set — STT disabled')
          return '[STT tidak tersedia — GOOGLE_APPLICATION_CREDENTIALS belum dikonfigurasi]'
        }

    voice.routes.ts:
      import multer from 'multer'
      const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 10 * 1024 * 1024 } })
      POST /transcribe → authMiddleware → requireRole('kader','ketua_kader') → upload.single('audio') → transcribeHandler
      transcribeHandler: if !req.file → 400 AUDIO_TIDAK_ADA; call transcribeAudio(req.file.buffer) → respond {transcript}

    useVoiceRecorder.ts (from 03-RESEARCH.md §Pattern 5 — full implementation provided):
      State: isRecording (boolean), audioBlob (Blob | null)
      Refs: mediaRecorderRef, chunksRef
      startRecording():
        Check mimeType support (CRITICAL — Safari compatibility from 03-RESEARCH.md §Pitfall 6):
          const mimeType = MediaRecorder.isTypeSupported('audio/webm;codecs=opus') ? 'audio/webm;codecs=opus'
                         : MediaRecorder.isTypeSupported('audio/mp4') ? 'audio/mp4' : ''
        navigator.mediaDevices.getUserMedia({ audio: true })
        new MediaRecorder(stream, { mimeType }) if mimeType else new MediaRecorder(stream)
        recorder.ondataavailable: push to chunksRef
        recorder.onstop: new Blob(chunks, {type:'audio/webm'}) → setAudioBlob; stop tracks
        recorder.start()
      stopRecording(): mediaRecorderRef.current?.stop(); setIsRecording(false)
      Return { isRecording, audioBlob, startRecording, stopRecording }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/voice/transcribe | Multipart audio blob from browser MediaRecorder |
| Browser → /api/ai/early-warning | pemeriksaanId + optional tandaKlinis |
| Backend → OpenAI API | System + user prompts containing balita name and Z-Scores |
| Backend → Google Cloud STT | Audio buffer (base64) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-06-01 | Denial of Service | Audio upload — very large file (100 MB) from malicious client | mitigate | Multer limits: { fileSize: 10 * 1024 * 1024 } — files > 10 MB rejected with 413 before reaching controller |
| T-03-06-02 | Tampering | OpenAI prompt injection via namaBalita or tanda klinis values | mitigate | namaBalita comes from DB (not raw from this request's body); the user prompt uses ${input.namaBalita} interpolation. System prompt hardcoded server-side. Risk low: even if injected, no function calling available to take unauthorized actions. |
| T-03-06-03 | Information Disclosure | OpenAI API key in browser | mitigate | OPENAI_API_KEY is process.env server-side only. generateEarlyWarning called in backend only. Frontend sends pemeriksaanId, never the key. |
| T-03-06-04 | Information Disclosure | Google Cloud credentials in browser | mitigate | GOOGLE_APPLICATION_CREDENTIALS is a file path on the server. SpeechClient initialized server-side. Audio blob is sent to backend then forwarded to Google. |
| T-03-06-05 | Information Disclosure | rekomendasiAi stored as plaintext in DB | mitigate | ai.routes.ts calls updatePemeriksaan(pemeriksaanId, {rekomendasiAi: result.rekomendasi}) — updatePemeriksaan encrypts via encrypt() before tx.pemeriksaan.update |
| T-03-06-06 | Spoofing + IDOR | pemeriksaanId in early-warning request belongs to different posyandu | mitigate | earlyWarningHandler fetches pemeriksaan from DB and verifies posyanduId matches kader's posyanduId before calling generateEarlyWarning |
| T-03-06-SC | Tampering | npm installs | mitigate | openai and @google-cloud/speech installed in 03-01 (pre-audited [OK]) |
</threat_model>
