        export const useKaderMejaStore = create<KaderMejaState>()(
          persist(

            { name: 'sispos-kader-meja', partialize: (s) => ({ isLocked: s.isLocked }) }
          )
        )

      CRITICAL: Do NOT persist activeMeja, activeSlotId to localStorage — Redis is the source of truth. These are restored from GET /api/kader/active-meja on mount.

    useActiveMeja.ts — TanStack Query, staleTime: 0, refetchOnMount: true:
      export function useActiveMeja() {
        return useQuery({
          queryKey: ['kader', 'active-meja'],
          queryFn: () => apiClient.get('/kader/active-meja').then(r => r.data.data),
          staleTime: 0,
          refetchOnMount: true,
        })
      }
      Also export useMutationSetActiveMeja() and useMutationClearActiveMeja() following useMutation pattern from 03-PATTERNS.md §usePemeriksaan hook.

        2. Guard: if (!slotId) return no-op

      Listen to 'queue:update' → invalidate ['antrian', 'kader', slotId]
