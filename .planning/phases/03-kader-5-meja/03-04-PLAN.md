    Create frontend-specific schemas using Zod v4 API in pemeriksaan.schemas.ts.

    pemeriksaan.schemas.ts (Zod v4 syntax):
      Meja2Schema = z.object({
        beratBadan: z.number({ error: 'Berat badan wajib diisi' }).positive('BB harus positif').max(100, 'BB maks 100 kg'),
        tinggiBadan: z.number().positive('TB harus positif').max(200, 'TB maks 200 cm').optional(),
        lingkarKepala: z.number().positive().optional(),
        lingkarLengan: z.number().positive().optional(),
        catatanKonsultasi: z.string().optional(),
      })
      export type Meja2Input = z.infer<typeof Meja2Schema>

    Note: In Zod v4, use error: not required_error. The BB max(100) allows 85 kg input — the biological validation gate is a UX confirmation, NOT a Zod error.

    usePemeriksaan.ts — follow useSesiAvailability.ts + useMutation pattern from 03-PATTERNS.md:

    usePemeriksaanHistory(balitaId: string | null):
      useQuery({
        queryKey: ['pemeriksaan', 'history', balitaId],
        queryFn: () => apiClient.get('/growth/balita/' + balitaId + '/history').then(r => r.data.data),
        enabled: !!balitaId,
        staleTime: 30_000,
      })

    useCreatePemeriksaan():
      useMutation({
        mutationFn: async (data: { balitaId: string, antrianId?: string, beratBadan: number, tinggiBadan?: number, konfirmasiBiologis?: boolean }) => {
          const headers: Record<string, string> = {}
          if (data.konfirmasiBiologis) headers['x-konfirmasi-biologis'] = 'true'
          return apiClient.post('/growth/pemeriksaan', data, { headers }).then(r => r.data.data)
        },
        onSuccess: (data, _vars, _ctx) => {
          void qc.invalidateQueries({ queryKey: ['pemeriksaan', 'history', data.balitaId] })
        },
      })

    usePatchPemeriksaan():
      useMutation({
        mutationFn: async ({ id, ...data }: { id: string, tandaKlinis?: object, statusGiziOverride?: string | null, catatanKlinis?: string, rekomendasiAi?: string, catatanKonsultasi?: string }) =>
          apiClient.patch('/growth/pemeriksaan/' + id, data).then(r => r.data.data),
        onSuccess: (data) => {
          void qc.invalidateQueries({ queryKey: ['pemeriksaan', 'history', data.balitaId] })
        },
      })

    Export all three hooks from usePemeriksaan.ts.
