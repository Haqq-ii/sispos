FILE: backend/src/modules/health/health.routes.ts
Create GET /api/health endpoint. Use the standard API response format from CLAUDE.md.
Import prisma from config/db, redis from config/redis, io from config/socket:

  import { Router } from 'express'
  import { prisma } from '../../config/db'
  import { redis } from '../../config/redis'

  export const healthRouter = Router()

  healthRouter.get('/', async (_req, res) => {
    let dbStatus = 'disconnected'
    let redisStatus = 'disconnected'
    let socketStatus = 'not_initialized'

    try {
      await prisma.$queryRaw`SELECT 1`
      dbStatus = 'connected'
    } catch { /* keep disconnected */ }

    try {
      const pong = await redis.ping()
      if (pong === 'PONG') redisStatus = 'connected'
    } catch { /* keep disconnected */ }

    // Socket.IO: io is set by initSocket in server.ts
    try {
      const { io } = await import('../../config/socket')
      socketStatus = io ? 'ready' : 'not_initialized'
    } catch { /* keep not_initialized */ }

    const allHealthy = dbStatus === 'connected' && redisStatus === 'connected' && socketStatus === 'ready'

    return res.status(allHealthy ? 200 : 503).json({
      success: allHealthy,
      data: { db: dbStatus, redis: redisStatus, socket: socketStatus },
      message: allHealthy ? 'Semua service berjalan normal' : 'Beberapa service belum siap'
    })
  })

Note: socketStatus check uses dynamic import to avoid circular dependency between socket.ts and health.routes.ts.
