    Step 2 — Create backend/src/modules/wilayah/wilayah.service.ts:

    Import prisma from backend/src/config/db.ts.

    Export async getProvinsi(): Promise<string[]>
      const rows = await prisma.wilayah.findMany({ distinct: ['provinsi'], select: { provinsi: true }, orderBy: { provinsi: 'asc' } })
      return rows.map(r => r.provinsi)

    Export async getKabupaten(provinsi: string): Promise<string[]>
      const rows = await prisma.wilayah.findMany({ where: { provinsi }, distinct: ['kabupaten'], select: { kabupaten: true }, orderBy: { kabupaten: 'asc' } })
      return rows.map(r => r.kabupaten)

    Export async getKecamatan(kabupaten: string, provinsi: string): Promise<string[]>
      const rows = await prisma.wilayah.findMany({ where: { kabupaten, provinsi }, distinct: ['kecamatan'], select: { kecamatan: true }, orderBy: { kecamatan: 'asc' } })
      return rows.map(r => r.kecamatan)

    Export async getKelurahan(kecamatan: string, kabupaten: string, provinsi: string): Promise<string[]>
      const rows = await prisma.wilayah.findMany({ where: { kecamatan, kabupaten, provinsi }, distinct: ['kelurahan'], select: { kelurahan: true }, orderBy: { kelurahan: 'asc' } })
      return rows.map(r => r.kelurahan)

    Step 3 — Create backend/src/modules/wilayah/wilayah.controller.ts:

    Export async function getProvinsiHandler(req: Request, res: Response): Promise<void>
      Call getProvinsi(). Return res.json({ success: true, data: result }).
      Wrap in try/catch — on error return 500 { success: false, error: 'INTERNAL_ERROR' }.

    Export async function getKabupatenHandler(req: Request, res: Response): Promise<void>
      Extract provinsi from req.query.provinsi as string. If !provinsi return 400 { error: 'PROVINSI_REQUIRED' }.
      Call getKabupaten(provinsi). Return 200.

    Export async function getKecamatanHandler(req: Request, res: Response): Promise<void>
      Extract kabupaten, provinsi from req.query. If missing return 400.
      Call getKecamatan(kabupaten, provinsi). Return 200.

    Export async function getKelurahanHandler(req: Request, res: Response): Promise<void>
      Extract kecamatan, kabupaten, provinsi from req.query. If missing return 400.
      Call getKelurahan(kecamatan, kabupaten, provinsi). Return 200.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → GET /api/wilayah/* | Public endpoint; query params are untrusted strings used in Prisma WHERE clauses |
| seed.wilayah.ts → prisma.wilayah.createMany | Seed data is hardcoded; no external input at seed time |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | GET /api/wilayah/kabupaten?provinsi=... SQL injection | mitigate | Prisma ORM parameterizes all WHERE clauses; no raw SQL; string input passed directly to Prisma where filter is safe |
| T-03-02 | Information Disclosure | Wilayah data sensitivity | accept | Geographic data (provinsi/kabupaten names) is public information; no PII involved |
| T-03-03 | Denial of Service | Mass concurrent requests to /api/wilayah/* | accept | Queries are lightweight DISTINCT selects with indexes on (provinsi, kabupaten) and (kecamatan, kelurahan); TanStack Query staleTime: 1hr on frontend dramatically reduces request volume |
| T-03-04 | Denial of Service | seed.wilayah.ts deleteMany wipes production data | mitigate | Seed script is development-only; never added to prisma.seed script for production; call explicitly via docker exec |
| T-03-SC | Tampering | No new npm packages in 01-03 | accept | No new packages installed; wilayah module uses only prisma + express already present |
</threat_model>
