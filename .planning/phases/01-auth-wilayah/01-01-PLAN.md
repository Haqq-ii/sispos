    Step 1 — Create backend/src/modules/auth/auth.service.ts with the following exported async functions. Do NOT use string concatenation for queries — use Prisma ORM exclusively.

    issueTokens(userId: string, role: string): { accessToken: string; refreshToken: string }
      Signs accessToken with jwt.sign({ userId, role }, env.JWT_SECRET, { expiresIn: env.JWT_ACCESS_EXPIRY }).
      Signs refreshToken with jwt.sign({ userId }, env.JWT_REFRESH_SECRET, { expiresIn: env.JWT_REFRESH_EXPIRY }).
      Returns both tokens as plain strings.

    registerWarga(data: z.infer<typeof RegisterSchema>): Promise<{ nomorPonselMasked: string }>
      Check unique: if (await prisma.warga.findUnique({ where: { nikIbu: data.nikIbu } })) throw error with code 'NIK_SUDAH_TERDAFTAR'.
      Check unique: if (await prisma.warga.findUnique({ where: { nomorPonsel: data.nomorPonsel } })) throw error with code 'HP_SUDAH_TERDAFTAR'.
      Hash password: bcrypt.hash(data.password, env.BCRYPT_ROUNDS).
      Create Warga: prisma.warga.create with nikIbu, namaLengkap, nomorPonsel, passwordHash, statusVerifikasi: 'belum_verifikasi'.
      Generate OTP: crypto.randomInt(100000, 999999).toString() — use Node's built-in crypto module.
      Expire time: new Date(Date.now() + 5 * 60 * 1000).
      Invalidate old OTPs: prisma.otp.updateMany({ where: { nomorPonsel, tujuan: 'registrasi', sudahDipakai: false }, data: { sudahDipakai: true } }).
      Save new Otp record: prisma.otp.create with nomorPonsel, kodeOtp, tujuan: 'registrasi', sudahDipakai: false, kedaluwarsaPada.
      Enqueue: await enqueueOtpJob(data.nomorPonsel, kodeOtp).
      Return masked phone: replace digits 4-8 with '****' (e.g., "081234567890" → "0812****7890").

    sendOtp(nomorPonsel: string): Promise<void>
      Find Warga by nomorPonsel — throw 'WARGA_TIDAK_DITEMUKAN' if not found.
      Generate OTP, invalidate old, save new, enqueue (same logic as registerWarga OTP section).

    verifyOtpAndLogin(nomorPonsel: string, kodeOtp: string): Promise<{ tokens: { accessToken: string; refreshToken: string }; user: { id: string; namaLengkap: string; role: 'citizen' } }>
      Find valid OTP: prisma.otp.findFirst({ where: { nomorPonsel, kodeOtp, sudahDipakai: false, kedaluwarsaPada: { gt: new Date() }, tujuan: 'registrasi' } }).
      If not found: throw error with code 'OTP_TIDAK_VALID'.
      Mark sudahDipakai: prisma.otp.update({ where: { id: otp.id }, data: { sudahDipakai: true } }).
      Update Warga status: prisma.warga.update({ where: { nomorPonsel }, data: { statusVerifikasi: 'terverifikasi' } }).
      Return issueTokens(warga.id, 'citizen') + user object { id, namaLengkap, role: 'citizen' }.

    updateLokasi(userId: string, data: z.infer<typeof UpdateLokasiSchema>): Promise<void>
      prisma.warga.update({ where: { id: userId }, data: { provinsi, kabupaten, kecamatan, kelurahan, rw: data.rw ?? null, rt: data.rt ?? null } }).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → POST /api/auth/register | Untrusted NIK, password, phone; validated by Zod before touching DB |
| POST /api/auth/otp/verify → DB | OTP code from client; must match DB record with TTL check |
| BullMQ worker → Fonnte API | WA delivery is external; failure triggers retry, never blocks registration |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | POST /api/auth/register | mitigate | Unique constraint on nikIbu + nomorPonsel; OTP verification required before terverifikasi status |
| T-01-02 | Tampering | POST /api/auth/otp/verify | mitigate | sudahDipakai flag set on first use; kedaluwarsaPada TTL 5 min; old OTPs invalidated on resend |
| T-01-03 | Information Disclosure | OTP enumeration via register/verify | mitigate | Response timing consistent for valid/invalid phone; masked phone in register response |
| T-01-04 | Denial of Service | POST /api/auth/otp/send repeated calls | mitigate | Max 3 resends per session (tracked in frontend + backend invalidates old OTPs); BullMQ queue absorbs load |
| T-01-05 | Information Disclosure | nomorPonsel in OTP job payload | accept | BullMQ job data in Redis, not persisted to DB; Redis requires AUTH per config |
| T-01-SC | Tampering | npm install jsonwebtoken, bcrypt, react-hook-form, zod, @hookform/resolvers | mitigate | Task 0 checkpoint requires human verification of publisher URLs before install; all from official npmjs.com registry |
</threat_model>
