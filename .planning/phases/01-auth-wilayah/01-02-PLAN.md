    login(identifier: string, password: string): Promise<{ tokens: ReturnType<typeof issueTokens>; user: AuthUser }>
      where AuthUser = { id: string; namaLengkap: string; role: RolePengguna }.
      Call detectRole(identifier). If null → throw { code: 'FORMAT_IDENTIFIER_TIDAK_VALID' }.

      Citizen path: prisma.warga.findUnique({ where: { nikIbu: identifier } }). If not found → throw { code: 'KREDENSIAL_SALAH' }. If statusVerifikasi !== 'terverifikasi' → throw { code: 'AKUN_BELUM_DIVERIFIKASI' }. bcrypt.compare(password, warga.passwordHash). If false → throw { code: 'KREDENSIAL_SALAH' }. Return issueTokens(warga.id, 'citizen') + user.

      Kader path: prisma.kader.findUnique({ where: { nomorPonsel: identifier } }). If not found → throw { code: 'KREDENSIAL_SALAH' }. If !kader.isAktif → throw { code: 'AKUN_TIDAK_AKTIF' }. Check terkunci: if (kader.terkunciSampai && kader.terkunciSampai > new Date()) throw { code: 'AKUN_TERKUNCI', terkunciSampai: kader.terkunciSampai }. bcrypt.compare(password, kader.pinHash). If false:
        const gagalBaru = kader.gagalLogin + 1
        if gagalBaru >= 10: prisma.kader.update({ data: { gagalLogin: gagalBaru, terkunciSampai: new Date(Date.now() + 30 * 60 * 1000) } })
        else: prisma.kader.update({ data: { gagalLogin: gagalBaru } })
        throw { code: 'KREDENSIAL_SALAH', gagalLogin: gagalBaru, maxGagal: 10 }
      If success: prisma.kader.update({ data: { gagalLogin: 0, terkunciSampai: null } }). Role = kader.isKetua ? 'ketua_kader' : 'kader'. Return issueTokens(kader.id, role) + user.

      Puskesmas path: prisma.puskesmas.findUnique({ where: { email: identifier } }). If not found → throw { code: 'KREDENSIAL_SALAH' }. bcrypt.compare(password, puskesmas.passwordHash). If false → throw { code: 'KREDENSIAL_SALAH' }. Return issueTokens(puskesmas.id, 'puskesmas') + user.

    refreshAccessToken(refreshToken: string): Promise<{ accessToken: string }>
      const decoded = jwt.verify(refreshToken, env.JWT_REFRESH_SECRET) as { userId: string }.
      Issue new accessToken: jwt.sign({ userId: decoded.userId }, env.JWT_SECRET, { expiresIn: env.JWT_ACCESS_EXPIRY }).
      Note: do NOT verify that user still exists in DB in Phase 1 (keep simple; add in hardening phase if needed).
      Return { accessToken }.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → POST /api/auth/login | Untrusted identifier + password/PIN; role detected server-side from format |
| access_token cookie → authMiddleware | JWT must be valid and signed with JWT_SECRET; forged tokens rejected |
| refresh_token cookie → POST /api/auth/refresh | Refresh token signed with separate JWT_REFRESH_SECRET |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | POST /api/auth/login — credential stuffing | mitigate | bcrypt cost factor 12 (slow hash); Kader 10-fail lockout 30 min (gagalLogin field); consistent error responses (no username enumeration beyond role format detection) |
| T-02-02 | Tampering | JWT access_token forging | mitigate | HS256 signed with JWT_SECRET (min 32 chars from env validation); authMiddleware rejects any invalid signature |
| T-02-03 | Information Disclosure | JWT payload inspection | mitigate | Stored as httpOnly cookie (JavaScript cannot read); never placed in localStorage or response body |
| T-02-04 | Elevation of Privilege | Citizen claiming kader role via modified JWT | mitigate | JWT role claim verified in authMiddleware on every request; Prisma queries use userId from verified token only |
| T-02-05 | Denial of Service | Brute-force PIN for locked Kader | mitigate | terkunciSampai checked before bcrypt compare — no CPU cost after lockout triggers |
| T-02-06 | Repudiation | Kader lock event untraced | accept | Phase 1: gagalLogin counter in DB is evidence; full AuditLog for MASTER_OVERRULE in Phase 4 (DSS-02) |
| T-02-SC | Tampering | No new npm packages in 01-02 | accept | No new packages installed; all dependencies from 01-01 already approved |
</threat_model>
